# megallm的请求方式
# 使用模型 ID 测试多个模型
for model in "gpt-5" "claude-opus-4-1-20250805" "gemini-2.5-pro"; do
  echo "测试 $model..."
  curl https://ai.megallm.io/v1/chat/completions \
    -H "Authorization: Bearer $API_KEY" \
    -H "Content-Type: application/json" \
    -d "{
      \"model\": \"$model\",
      \"messages\": [{\"role\": \"user\", \"content\": \"你好！\"}]
    }"
done
